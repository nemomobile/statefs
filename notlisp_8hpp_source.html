400 template <typename ArgFnT, typename KeyFnT>
401 void rest(ListAccessor &src, ArgFnT arg, KeyFnT kwd_ard)
402 {
403  expr_ptr k, v;
404  while (src.optional(v)) {
405  if (!k) {
406  if (v && v->type() == Expr::Keyword)
407  k = v;
408  else
409  arg(v);
410  } else {
411  kwd_ard(k, v);
412  k = nullptr;
413  }
414  }
415  if (k)
416  throw Error("Orphaned keyword");
417 }
